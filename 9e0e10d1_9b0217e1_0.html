【1】Go在不同类型的变量之间赋值时需要显式转换，并且只有显式转换(强制转换)。
【2】语法：
表达式T(v)将值v转换为类型T
T : 就是数据类型
v : 就是需要转换的变量

【3】案例展示：  


  1. package main
  2. import "fmt"
  4. func main(){
  5.         //进行类型转换：
  6.         var n1 int = 100
  7.         //var n2 float32 = n1  在这里自动转换不好使，比如显式转换
  8.         fmt.Println(n1)
  9.         //fmt.Println(n2)
  10.         var n2 float32 = float32(n1)
  11.         fmt.Println(n2)
  12.         //注意：n1的类型其实还是int类型，只是将n1的值100转为了float32而已，n1还是int的类型
  13.         fmt.Printf("%T",n1)  //int
  14.         fmt.Println()
  16.         //将int64转为int8的时候，编译不会出错的，但是会数据的溢出
  17.         var n3 int64 = 888888
  18.         var n4 int8 = int8(n3)
  19.         fmt.Println(n4)//56
  21.         var n5 int32 = 12
  22.         var n6 int64 = int64(n5) + 30  //一定要匹配=左右的数据类型
  23.         fmt.Println(n5)
  24.         fmt.Println(n6)
  27.         var n7 int64 = 12
  28.         var n8 int8 = int8(n7) + 127  //编译通过，但是结果可能会溢出
  29.         //var n9 int8 = int8(n7) + 128 //编译不会通过
  30.         fmt.Println(n8)
  31.         //fmt.Println(n9)
  34. }
