【1】算术运算符：+ ，-，*，/，%，++，--
【2】介绍：算术运算符是对数值类型的变量进行运算的，比如，加减乘除。
【3】代码展示： 

  1. package main
  2. import "fmt"
  3. func main(){
  4.         //+加号：
  5.         //1.正数 2.相加操作  3.字符串拼接
  6.         var n1 int = +10
  7.         fmt.Println(n1)
  8.         var n2 int = 4 + 7
  9.         fmt.Println(n2)
  10.         var s1 string = "abc" + "def"
  11.         fmt.Println(s1)
  13.         // /除号：
  14.         fmt.Println(10/3) //两个int类型数据运算，结果一定为整数类型
  15.         fmt.Println(10.0/3)//浮点类型参与运算，结果为浮点类型
  17.         // % 取模  等价公式： a%b=a-a/b*b
  18.         fmt.Println(10%3) // 10%3= 10-10/3*3 = 1
  19.         fmt.Println(-10%3)
  20.         fmt.Println(10%-3)
  21.         fmt.Println(-10%-3)
  23.         //++自增操作：
  24.         var a int = 10
  25.         a++
  26.         fmt.Println(a)
  27.         a--
  28.         fmt.Println(a)
  29.         //++ 自增 加1操作，--自减，减1操作
  30.         //go语言里，++，--操作非常简单，只能单独使用，不能参与到运算中去
  31.         //go语言里，++，--只能在变量的后面，不能写在变量的前面 --a  ++a  错误写法
  32. }
