1. package main
  2. import "fmt"
  3. func main(){
  4.         //方式1：
  5.         //定义map变量：
  6.         var a map[int]string
  7.         //只声明map内存是没有分配空间
  8.         //必须通过make函数进行初始化，才会分配空间：
  9.         a = make(map[int]string,10) //map可以存放10个键值对
  10.         //将键值对存入map中：
  11.         a[20095452] = "张三"
  12.         a[20095387] = "李四"
  13.         //输出集合
  14.         fmt.Println(a)
  16.         //方式2：
  17.         b := make(map[int]string)
  18.         b[20095452] = "张三"
  19.         b[20095387] = "李四"
  20.         fmt.Println(b)
  22.         //方式3：
  23.         c := map[int]string{
  24.                 20095452 : "张三",
  25.                 20098765 : "李四",
  26.         }
  27.         c[20095387] = "王五"
  28.         fmt.Println(c)
  29. }
