【1】continue的作用： 

  1. package main
  2. import "fmt"
  3. func main(){
  4.         //功能：输出1-100中被6整除的数：
  5.         //方式1：
  6.         // for i := 1; i <= 100; i++ {
  7.         // 	if i % 6 == 0 {
  8.         // 		fmt.Println(i)
  9.         // 	}
  10.         // }
  12.         //方式2：
  13.         for i := 1; i <= 100; i++ {
  14.                 if i % 6 != 0 {
  15.                         continue //结束本次循环，继续下一次循环
  16.                 }
  17.                 fmt.Println(i)
  18.         }
  19. }
【2】深入理解： 

  1. package main
  2. import "fmt"
  3. func main(){
  4.         //双重循环：
  5.         for i := 1; i <= 5; i++ {
  6.                 for j := 2; j <= 4; j++ {			
  7.                         if i == 2 && j == 2 {
  8.                                 continue
  9.                         }
  10.                         fmt.Printf("i: %v, j: %v \n",i,j)
  11.                 }
  12.         }
  13.         fmt.Println("-----ok")
  14. }
结果：

结论：continue的作用是结束离它近的那个循环，继续离它近的那个循环

【3】标签的作用： 

  1. package main
  2. import "fmt"
  3. func main(){
  4.         //双重循环：
  5.         label:
  6.         for i := 1; i <= 5; i++ {
  7.                 for j := 2; j <= 4; j++ {			
  8.                         if i == 2 && j == 2 {
  9.                                 continue label
  10.                         }
  11.                         fmt.Printf("i: %v, j: %v \n",i,j)
  12.                 }
  13.         }
  14.         fmt.Println("-----ok")
  15. }
