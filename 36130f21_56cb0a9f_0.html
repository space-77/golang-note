【1】为什么要使用函数：
提高代码的复用型，减少代码的冗余，代码的维护性也提高了

【2】函数的定义：
为完成某一功能的程序指令(语句)的集合,称为函数。

【3】基本语法
func   函数名（形参列表)（返回值类型列表）{
      执行语句..
      return + 返回值列表
}

【4】函数的定义和函数的调用案例： 


  1. package main
  2. import "fmt"
  4. // func   函数名（形参列表)（返回值类型列表）{
  5. // 	执行语句..
  6. // 	return + 返回值列表
  7. // }
  9. //自定义函数：功能：两个数相加：
  10. func cal (num1 int,num2 int) (int) { //如果返回值类型就一个的话，那么()是可以省略不写的
  11.         var sum int = 0
  12.         sum += num1
  13.         sum += num2
  14.         return sum
  15. }
  18. func main(){
  19.         //功能：10 + 20
  20.         //调用函数：
  21.         sum := cal(10,20)
  22.         fmt.Println(sum)
  23.         // var num1 int = 10
  24.         // var num2 int = 20

  31.         //功能：30 + 50

  33.         var num4 int = 50

  35.         sum1 := cal(num3,num4)

  37.         //求和：
  38.         // var sum1 int = 0
