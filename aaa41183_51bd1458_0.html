【1】基本数据类型和string的转换介绍
在程序开发中，我们经常需要将基本数据类型转成string类型。或者将string类型转成基本数据类型。
【2】基本类型转string类型
方式1:fmt.Sprintf("%参数",表达式)    ---》 重点练习这个，推荐方式
方式2:使用strconv包的函数   

【3】代码测试：


方式1： 
  1. package main
  2. import "fmt"
  4. func main(){
  5.         var n1 int = 19
  6.         var n2 float32 = 4.78
  7.         var n3 bool = false
  8.         var n4 byte = 'a'
  10.         var s1 string = fmt.Sprintf("%d",n1)
  11.         fmt.Printf("s1对应的类型是：%T ，s1 = %q \n",s1, s1)
  13.         var s2 string = fmt.Sprintf("%f",n2)
  14.         fmt.Printf("s2对应的类型是：%T ，s2 = %q \n",s2, s2)
  16.         var s3 string = fmt.Sprintf("%t",n3)
  17.         fmt.Printf("s3对应的类型是：%T ，s3 = %q \n",s3, s3)
  19.         var s4 string = fmt.Sprintf("%c",n4)
  20.         fmt.Printf("s4对应的类型是：%T ，s4 = %q \n",s4, s4)
  21. }
方式2： 


  1. package main
  2. import(
  3.         "fmt"
  4.         "strconv"
  5. )
  7. func main(){
  8.         var n1 int = 18
  9.         var s1 string = strconv.FormatInt(int64(n1),10)  //参数：第一个参数必须转为int64类型 ，第二个参数指定字面值的进制形式为十进制
  10.         fmt.Printf("s1对应的类型是：%T ，s1 = %q \n",s1, s1)
  12.         var n2 float64 = 4.29
  13.         var s2 string = strconv.FormatFloat(n2,'f',9,64)
  14.         //第二个参数：'f'（-ddd.dddd）  第三个参数：9 保留小数点后面9位  第四个参数：表示这个小数是float64类型
  15.         fmt.Printf("s2对应的类型是：%T ，s2 = %q \n",s2, s2)
  17.         var n3 bool = true
  18.         var s3 string = strconv.FormatBool(n3)
  19.         fmt.Printf("s3对应的类型是：%T ，s3 = %q \n",s3, s3)
  21. }
