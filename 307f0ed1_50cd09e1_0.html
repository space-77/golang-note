【1】方式1：for循环常规方式遍历
【2】方式2：for-range 结构遍历切片 


  1. package main
  2. import "fmt"
  3. func main(){
  4.         //定义切片：
  5.         slice := make([]int,4,20)
  6.         slice[0] = 66
  7.         slice[1] = 88
  8.         slice[2] = 99
  9.         slice[3] = 100
  10.         //方式1：普通for循环
  11.         for i := 0;i < len(slice);i++ {
  12.                 fmt.Printf("slice[%v] = %v \t" ,i,slice[i])
  13.         }
  14.         fmt.Println("\n------------------------------")
  15.         //方式2：for-range循环：
  16.         for i,v := range slice {
  17.                 fmt.Printf("下标：%v ，元素：%v\n" ,i,v)
  18.         }
  19. }
