【1】赋值运算符:=,+=，-=，*=，/=,%=
【2】赋值运算符就是将某个运算后的值，赋给指定的变量。
【3】代码展示： 

  1. package main
  2. import "fmt"
  3. func main(){
  4.         var num1 int = 10
  5.         fmt.Println(num1)
  6.         var num2 int = (10 + 20) % 3 + 3 - 7   //=右侧的值运算清楚后，再赋值给=的左侧
  7.         fmt.Println(num2)
  9.         var num3 int = 10
  10.         num3 += 20 //等价num3 = num3 + 20;
  11.         fmt.Println(num3)
  12. }
