【1】普通for循环
【2】键值循环（for range）


代码： 

  1. package main
  2. import "fmt"
  3. func main(){
  4.         //定义二维数组：
  5.         var arr [3][3]int = [3][3]int{{1,4,7},{2,5,8},{3,6,9}}
  6.         fmt.Println(arr)
  7.         fmt.Println("------------------------")
  8.         //方式1：普通for循环：
  9.         for i := 0;i < len(arr);i++{
  10.                 for j := 0;j < len(arr[i]);j++ {
  11.                         fmt.Print(arr[i][j],"\t")
  12.                 }
  13.                 fmt.Println()
  14.         }
  15.         fmt.Println("------------------------")
  16.         //方式2：for range循环：
  17.         for key,value := range arr {
  18.                 for k,v := range value {
  19.                         fmt.Printf("arr[%v][%v]=%v\t",key,k,v)
  20.                 }
  21.                 fmt.Println()
  22.         }
  23. }
