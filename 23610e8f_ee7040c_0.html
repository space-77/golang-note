【5】变量的4种使用方式：
【6】支持一次性声明多个变量（多变量声明） 
  1. package main
  2. import "fmt"
  4. //全局变量：定义在函数外的变量
  5. var n7 = 100
  6. var n8 = 9.7
  8. //设计者认为上面的全局变量的写法太麻烦了，可以一次性声明：
  9. var (
  10.         n9 = 500
  11.         n10 = "netty"
  12. )
  15. func main(){
  16.         //定义在{}中的变量叫：局部变量
  17.         //第一种：变量的使用方式：指定变量的类型，并且赋值，
  18.         var num int = 18
  19.         fmt.Println(num)
  21.         //第二种：指定变量的类型，但是不赋值，使用默认值 
  22.         var num2 int
  23.         fmt.Println(num2)
  25.         //第三种：如果没有写变量的类型，那么根据=后面的值进行判定变量的类型 （自动类型推断）
  26.         var num3 = "tom"
  27.         fmt.Println(num3)
  29.         //第四种：省略var，注意 := 不能写为 =   
  30.         sex := "男"
  31.         fmt.Println(sex)
  33.         fmt.Println("------------------------------------------------------------------")
  34.         //声明多个变量：
  35.         var n1,n2,n3 int
  36.         fmt.Println(n1)
  37.         fmt.Println(n2)
  38.         fmt.Println(n3)
  40.         var n4,name,n5 = 10,"jack",7.8
  41.         fmt.Println(n4)
  42.         fmt.Println(name)
  43.         fmt.Println(n5)
  45.         n6,height := 6.9,100.6
  46.         fmt.Println(n6)
  47.         fmt.Println(height)
  50.         fmt.Println(n7)
  51.         fmt.Println(n8)
  53.         fmt.Println(n9)
  54.         fmt.Println(n10)
  56. }
