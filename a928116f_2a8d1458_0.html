【1】string类型转基本类型
方式:使用strconv包的函数   
【2】测试： 


  1. package main
  2. import(
  3.         "fmt"
  4.         "strconv"
  5. )
  7. func main(){
  8.         //string-->bool
  9.         var s1 string = "true"
  10.         var b bool
  11.         //ParseBool这个函数的返回值有两个：(value bool, err error)
  12.         //value就是我们得到的布尔类型的数据，err出现的错误
  13.         //我们只关注得到的布尔类型的数据，err可以用_直接忽略
  14.         b , _ = strconv.ParseBool(s1)
  15.         fmt.Printf("b的类型是：%T,b=%v \n",b,b)
  17.         //string---》int64
  18.         var s2 string = "19"
  19.         var num1 int64
  20.         num1,_ = strconv.ParseInt(s2,10,64)
  21.         fmt.Printf("num1的类型是：%T,num1=%v \n",num1,num1)
  23.         //string-->float32/float64
  24.         var s3 string = "3.14"
  25.         var f1 float64
  26.         f1,_ = strconv.ParseFloat(s3,64)
  27.         fmt.Printf("f1的类型是：%T,f1=%v \n",f1,f1)
  29.         //注意：string向基本数据类型转换的时候，一定要确保string类型能够转成有效的数据类型，否则最后得到的结果就是按照对应类型的默认值输出
  30.         var s4 string = "golang"
  31.         var b1 bool
  32.         b1 , _ = strconv.ParseBool(s4)
  33.         fmt.Printf("b1的类型是：%T,b1=%v \n",b1,b1)
  35.         var s5 string = "golang"
  36.         var num2 int64
  37.         num2,_ = strconv.ParseInt(s5,10,64)
  38.         fmt.Printf("num2的类型是：%T,num2=%v \n",num2,num2)
  39. }
