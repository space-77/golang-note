【1】其它运算符：
& ：返回变量的存储地址 
*：取指针变量对应的数值


【2】代码练习： 

  1. package main
  2. import "fmt"
  3. func main(){
  4.         //定义一个变量：
  5.         var age int = 18
  6.         fmt.Println("age对应的存储空间的地址为：",&age)//age对应的存储空间的地址为： 0xc0000100b0
  8.         var ptr *int = &age
  9.         fmt.Println(ptr)
  10.         fmt.Println("ptr这个指针指向的具体数值为：",*ptr)
  13. }
